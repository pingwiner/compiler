    a = f1();

fun _div(x, y) {
    retVal = 0;
    {
        x = x - y;
        retVal = retVal + 1;
    } while (x >= y);
    return retVal;
}

fun _mul(x, y) {
    result = 0;
    {
        result = result + x;
        y = y - 1;
    } while(y > 0);
    return result;
}
