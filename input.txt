fun _div(x, y) {
    _R0 = 0;
    {
        x = x - y;
        _R0 = _R0 + 1;
    } while (x >= y);
    return _R0;
}

fun _mul(x, y) {
    _R0 = 0;
    {
        _R0 = _R0 + x;
        y = y - 1;
    } while(y > 0);
    return _R0;
}

fun _mod(x, y) {
    _R0 = x;
    {
        _R0 = _R0 - y;
    } while (_R0 >= y);
    return _R0;
}
